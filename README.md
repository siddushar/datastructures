# datastructures
# Linked List
    # Complexity
        prepend - O(1)
        append  - O(1)
        lookup  - O(n)
        insert  - O(n)
        delete  - O(n)
     
    # Advantages
      1. Fast insertion
      2. Fast deletion
      3. Ordered
      4. Flexible size
   
    # Disdvantages
      1. Slow lookup
      2. More memory

# Stacks
    # Complexity
        push   - O(1)
        pop    - O(1)
        lookup - O(n)
        peek   - O(1)
        
        
# Queues
    # Complexity
        enquque - O(1) # push
        dequque - O(1) # pop
        lookup  - O(n)
        peek    - O(1)
